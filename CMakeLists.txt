CMAKE_MINIMUM_REQUIRED(VERSION 3.0 FATAL_ERROR)
PROJECT (Neutron_background_study)

LIST(APPEND CMAKE_PREFIX_PATH $ENV{ROOTSYS})

FIND_PACKAGE(ROOT REQUIRED COMPONENTS RIO Net)

INCLUDE(${ROOT_USE_FILE})

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/include)

ADD_LIBRARY(bkg SHARED ${CMAKE_CURRENT_SOURCE_DIR}/src/background.cpp)
TARGET_LINK_LIBRARIES(bkg ${ROOT_LIBRARIES})

ADD_EXECUTABLE(BackgroundAnalysis ${CMAKE_CURRENT_SOURCE_DIR}/src/background.cpp)
TARGET_LINK_LIBRARIES(BackgroundAnalysis bkg)
